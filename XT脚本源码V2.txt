local function LoadXT_Script()
    -- 添加详细的加载状态反馈
    game:GetService("StarterGui"):SetCore("SendNotification", {
        Title = "XT脚本",
        Text = "开始加载脚本...",
        Duration = 2
    })

    wait(1)

    local success, err = pcall(function()
        -- 使用原始GitHub链接加载脚本
        local scriptUrl = "https://raw.githubusercontent.com/XTxt-design/XT-/refs/heads/mai"
        local scriptContent = game:HttpGet(scriptUrl, true)

        -- 检查脚本内容是否有效
        if not scriptContent or #scriptContent < 100 then
            error("无法获取脚本内容或内容过短")
        end

        -- 编译并执行脚本
        local loadFunc, compileError = loadstring(scriptContent)
        if not loadFunc then
            error("脚本编译错误: "..tostring(compileError))
        end

        loadFunc()
    end)

    if not success then
        warn("[XT脚本] 加载错误: "..tostring(err))
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "XT脚本错误",
            Text = "加载失败: "..tostring(err),
            Duration = 10
        })
    else
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "XT脚本",
            Text = "加载成功!",
            Duration = 3
        })
    end
end

-- 延迟加载确保注入器完全初始化

local LastIteration, Start
local FrameUpdateTable = { }

local function HeartbeatUpdate()
	LastIteration = tick()
	for Index = #FrameUpdateTable, 1, -1 do
		FrameUpdateTable[Index + 1] = (FrameUpdateTable[Index] >= LastIteration - 1) and FrameUpdateTable[Index] or nil
	end
	FrameUpdateTable[1] = LastIteration
	local CurrentFPS = (tick() - Start >= 1 and #FrameUpdateTable) or (#FrameUpdateTable / (tick() - Start))
	CurrentFPS = CurrentFPS - CurrentFPS % 1
	FpsLabel.Text = ("当前时间:"..os.date("%H").."时"..os.date("%M").."分"..os.date("%S"))
end
Start = tick()
